julia> function simulation(a,Yt,b,q,T,Xo)
         xnew = []
         for i in 1:T
           push!(xnew, Xo) 
           Xt = Xo + a + Yt[i] + (Xo^q)/(1 + Xo^q) - b*Xo 
           Xo = Xt
         end
         return xnew
       end
